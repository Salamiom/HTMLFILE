class Bitwise
{
public static void main(String args[])
{
int a =5, b = 7;
 int res = a&b;
 System.out.println("the result is" +res);
 
 int res = a|b;
 System.out.println("the result is" +res);
 
 int res = a^b;
 System.out.println("the result is" +res);
 
 int res = a<<b;
 System.out.println("the result is" +res);
 
 int res = a>>b;
 System.out.println("the result is" +res);
 
 }}
